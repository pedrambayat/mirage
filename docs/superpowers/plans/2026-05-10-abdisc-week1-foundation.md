"""Core abstractions for scorers and the benchmark examples they consume."""
from __future__ import annotations

from dataclasses import dataclass, field
from typing import Any, Literal

ExampleClass = Literal["POS", "OFF", "SCR", "DES"]
_VALID_CLASSES: frozenset[str] = frozenset({"POS", "OFF", "SCR", "DES"})


@dataclass(frozen=True)
class BenchmarkExample:
    """One row of the benchmark — a binder-target pair with provenance."""

    id: str
    antibody_seq: str
    antigen_seq: str
    klass: str  # one of POS, OFF, SCR, DES
    antigen_pdb_id: str | None = None
    complex_pdb_id: str | None = None
    source: str = "unknown"
    metadata: dict[str, Any] = field(default_factory=dict)

    def __post_init__(self) -> None:
        if self.klass not in _VALID_CLASSES:
            raise ValueError(
                f"klass must be one of {sorted(_VALID_CLASSES)}, got {self.klass!r}"
            )


@dataclass(frozen=True)
class Score:
    """Output of a scorer on one BenchmarkExample."""

    value: float
    scorer_name: str
    example_id: str
    metadata: dict[str, Any] = field(default_factory=dict)
```

from abc import ABC, abstractmethod


class AbstractScorer(ABC):
    """Base class every scorer must implement.

    Subclasses set the class attribute `name` (a stable string id used
    in leaderboard tables and CLI dispatch) and implement `score`.
    """

    name: str = ""

    def __init__(self) -> None:
        if not self.name:
            raise NotImplementedError(
                f"{type(self).__name__} must set class attribute `name`."
            )

    @abstractmethod
    def score(self, example: BenchmarkExample) -> Score:
        """Score one example. Higher = more likely to be a true binder."""
```

"""A small in-process registry mapping scorer name → class.

Scorer modules call `register_scorer(REGISTRY)` as a class decorator.
The CLI imports the registry to dispatch by name.
"""
from __future__ import annotations

from typing import Callable, TypeVar

from abdisc.scorers.base import AbstractScorer

_T = TypeVar("_T", bound=type[AbstractScorer])


class ScorerRegistry:
    def __init__(self) -> None:
        self._scorers: dict[str, type[AbstractScorer]] = {}

    def register(self, cls: type[AbstractScorer]) -> None:
        if not cls.name:
            raise ValueError(f"{cls.__name__} has no `name` attribute set.")
        if cls.name in self._scorers:
            raise ValueError(f"Scorer {cls.name!r} already registered.")
        self._scorers[cls.name] = cls

    def get(self, name: str) -> type[AbstractScorer]:
        if name not in self._scorers:
            raise KeyError(name)
        return self._scorers[name]

    def list_names(self) -> list[str]:
        return sorted(self._scorers)


REGISTRY = ScorerRegistry()


def register_scorer(registry: ScorerRegistry) -> Callable[[_T], _T]:
    """Class decorator that registers a scorer class with `registry`."""

    def _decorator(cls: _T) -> _T:
        registry.register(cls)
        return cls

    return _decorator
```

"""Typer entry point: `abdisc` command."""
from __future__ import annotations

import typer

from abdisc import __version__
from abdisc.scorers._registry import REGISTRY

app = typer.Typer(help="abdisc — antibody-antigen discrimination benchmark.")
scorers_app = typer.Typer(help="Inspect registered scorers.")
app.add_typer(scorers_app, name="scorers")


def _version_callback(value: bool) -> None:
    if value:
        typer.echo(__version__)
        raise typer.Exit()


@app.callback()
def root(
    version: bool = typer.Option(
        False,
        "--version",
        callback=_version_callback,
        is_eager=True,
        help="Print version and exit.",
    ),
) -> None:
    """Top-level `abdisc` command."""


@scorers_app.command("list")
def scorers_list() -> None:
    """List all registered scorers."""
    names = REGISTRY.list_names()
    if not names:
        typer.echo("No scorers registered.")
        return
    typer.echo("Registered scorers:")
    for n in names:
        typer.echo(f"  - {n}")
```
